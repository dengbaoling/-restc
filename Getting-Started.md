<!-- # 用法 -->

### Node (CommonJS)

```shell
# 安装
npm install mockjs
```
   
```js
// 使用 Mock
var Mock = require('mockjs')
var data = Mock.mock({
    'list|1-10': [{
        'id|+1': 1
    }]
})
// 输出结果
console.log(JSON.stringify(data, null, 4))
```

### RequireJS (AMD)

```js
// 配置 Mock 路径
require.config({
  paths:{
    'mock':'http://mockjs.com/dist/mock'
  }
})
// 加载 Mock
require(['mock'], function(Mock){
    // 使用 Mock
    var data = Mock.mock({
        'list|1-10': [{
            'id|+1': 1
        }]
    })
    // 输出结果
    $('<pre>').text(JSON.stringify(data, null, 4))
        .appendTo('body')
})
```

[JSFiddle](http://jsfiddle.net/uTSqT/3/)

### Sea.js (CMD)

```js
// 配置 Mock 路径
seajs.config({
  alias: {
    "mock": "http://mockjs.com/dist/mock.js"
  }
})

// 加载 Mock
seajs.use('mock', function(Mock){
    // 使用 Mock
    var data = Mock.mock({
        'list|1-10': [{
            'id|+1': 1
        }]
    })
    // 输出结果
    $('<pre>').text(JSON.stringify(data, null, 4))
        .appendTo('body')
})
```
